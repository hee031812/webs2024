$(function(){
    let currentIndex = 0;   // 현재 이미지

    setInterval(function(){             //3초에 한번씩 실행.
        let nextIndex = (currentIndex +1) % 5;

        $(".slider").eq(currentIndex).fadeOut(1200);        // 첫번째 이미지 나타남
        $(".slider").eq(nextIndex).fadeIn(1200);            // 두번째 이미지 나타남

        currentIndex = nextIndex;

    }, 3000);
});
